function createCounter() {
  let i = 0;

  return {
    increment: () => {
      return i++;
    },
    decrement:()=>{
      return i--
    }
  };
}

const counter = createCounter();
console.log("hi: ", counter.increment()); 
console.log("hi: ", counter.increment()); // 1
console.log("hi: ", counter.decrement()); // 2
console.log("hi: ", counter.increment()); // 3
